**1. Create `backend/app/services/feishu_client.py`** — thin httpx wrapper (per research pattern):

```python
from __future__ import annotations

import time
from typing import Optional

import httpx

FEISHU_BASE_URL = "https://open.feishu.cn/open-apis"


class FeishuClient:
    """Low-level Feishu API client with automatic tenant_access_token management."""

    def __init__(self, app_id: str, app_secret: str):
        self._app_id = app_id
        self._app_secret = app_secret
        self._token: Optional[str] = None
        self._token_expires_at: float = 0
        self._http = httpx.Client(base_url=FEISHU_BASE_URL, timeout=30)

    def _ensure_token(self) -> str:
        """Get valid tenant_access_token, refreshing if expired (300s buffer)."""
        if self._token and time.time() < self._token_expires_at - 300:
            return self._token
        resp = self._http.post("/auth/v3/tenant_access_token/internal", json={
            "app_id": self._app_id,
            "app_secret": self._app_secret,
        })
        resp.raise_for_status()
        data = resp.json()
        if data.get("code") != 0:
            raise FeishuApiError(data.get("msg", "Failed to get tenant_access_token"), data.get("code"))
        self._token = data["tenant_access_token"]
        self._token_expires_at = time.time() + data["expire"]
        return self._token

    def _headers(self) -> dict[str, str]:
        return {"Authorization": f"Bearer {self._ensure_token()}"}

    def list_fields(self, app_token: str, table_id: str) -> list[dict]:
        """Fetch all field definitions from a Bitable table."""
        items = []
        page_token: Optional[str] = None
        while True:
            params: dict = {"page_size": 100}
            if page_token:
                params["page_token"] = page_token
            resp = self._http.get(
                f"/bitable/v1/apps/{app_token}/tables/{table_id}/fields",
                headers=self._headers(), params=params
            )
            resp.raise_for_status()
            data = resp.json()
            if data.get("code") != 0:
                raise FeishuApiError(data.get("msg", "list_fields failed"), data.get("code"))
            items.extend(data["data"]["items"])
            if not data["data"].get("has_more"):
                break
            page_token = data["data"].get("page_token")
        return items

    def batch_create_records(self, app_token: str, table_id: str, records: list[dict]) -> dict:
        """Create records in batch. Max 500 per call (conservative, API limit 1000)."""
        resp = self._http.post(
            f"/bitable/v1/apps/{app_token}/tables/{table_id}/records/batch_create",
            headers=self._headers(),
            json={"records": [{"fields": r} for r in records]}
        )
        resp.raise_for_status()
        data = resp.json()
        if data.get("code") != 0:
            raise FeishuApiError(data.get("msg", "batch_create failed"), data.get("code"))
        return data

    def batch_update_records(self, app_token: str, table_id: str, records: list[dict]) -> dict:
        """Update records in batch. Each record must have 'record_id' and 'fields'."""
        resp = self._http.post(
            f"/bitable/v1/apps/{app_token}/tables/{table_id}/records/batch_update",
            headers=self._headers(),
            json={"records": records}
        )
        resp.raise_for_status()
        data = resp.json()
        if data.get("code") != 0:
            raise FeishuApiError(data.get("msg", "batch_update failed"), data.get("code"))
        return data

    def search_records(self, app_token: str, table_id: str,
                       filter_expr: Optional[str] = None,
                       page_token: Optional[str] = None,
                       page_size: int = 500) -> dict:
        """Search records with optional filter. Max 500 per page."""
        body: dict = {"page_size": page_size}
        if filter_expr:
            body["filter"] = filter_expr
        if page_token:
            body["page_token"] = page_token
        resp = self._http.post(
            f"/bitable/v1/apps/{app_token}/tables/{table_id}/records/search",
            headers=self._headers(), json=body
        )
        resp.raise_for_status()
        data = resp.json()
        if data.get("code") != 0:
            raise FeishuApiError(data.get("msg", "search_records failed"), data.get("code"))
        return data

    def close(self):
        self._http.close()


class FeishuApiError(Exception):
    def __init__(self, message: str, code: int = 0):
        super().__init__(message)
        self.code = code
```

from typing import Optional

from backend.app.core.config import Settings

_client_instance: Optional[FeishuClient] = None


def get_feishu_client(settings: Optional[Settings] = None) -> FeishuClient:
    """Return a shared FeishuClient instance. Raises ValueError if credentials not configured."""
    global _client_instance
    if _client_instance is not None:
        return _client_instance
    if settings is None:
        from backend.app.core.config import get_settings
        settings = get_settings()
    if not settings.feishu_app_id or not settings.feishu_app_secret:
        raise ValueError("Feishu credentials not configured (FEISHU_APP_ID / FEISHU_APP_SECRET)")
    _client_instance = FeishuClient(settings.feishu_app_id, settings.feishu_app_secret)
    return _client_instance


def reset_feishu_client():
    """Reset the singleton (for testing or credential update)."""
    global _client_instance
    if _client_instance is not None:
        _client_instance.close()
    _client_instance = None
