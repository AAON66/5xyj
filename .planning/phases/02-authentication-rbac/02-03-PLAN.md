    - Add new function `verifyEmployee`:
      ```typescript
      export interface EmployeeVerifyInput {
        employee_id: string;
        id_number: string;
        person_name: string;
      }

      export async function verifyEmployee(input: EmployeeVerifyInput): Promise<AuthSession> {
        const response = await apiClient.post<ApiSuccessResponse<AuthLoginResponse>>('/auth/employee-verify', input);
        const payload = response.data.data;
        return {
          accessToken: payload.access_token,
          expiresAt: payload.expires_at,
          username: payload.user.username,
          role: payload.user.role as AuthRole,
          displayName: payload.user.display_name,
          signedInAt: new Date().toISOString(),
          mustChangePassword: false,
        };
      }
      ```
    - Update `AuthLoginResponse` interface to include `must_change_password?: boolean` in the `user` field

    **3. Update authContext.ts:**
    - Update `AuthenticatedUser` to include `mustChangePassword?: boolean`
    - Add `verifyEmployee` to `AuthContextValue`: `verifyEmployee: (input: EmployeeVerifyInput) => Promise<void>`
    - Import `EmployeeVerifyInput` from auth service

    **4. Update AuthProvider.tsx:**
    - Add `verifyEmployee` function alongside existing `login`:
      ```typescript

        const nextSession = await verifyEmployeeApi(input);

        setSession(nextSession);
      }
      ```
    - Import `verifyEmployee as verifyEmployeeApi` from auth service
    - Add `verifyEmployee: handleVerifyEmployee` to the context value
