**步骤 3：重写 buildMenuItems 函数**

新函数签名保持 `buildMenuItems`，但输入改为分组结构：

```typescript
function buildMenuItems(
  topItem: NavItem,
  groups: MenuGroupConfig[],
  feishuItems: NavItem[],
  userRole: string
): MenuProps['items'] {
  // 1. 员工角色（per IC-07）：不分组，只返回 employee 可见的单项
  if (userRole === 'employee') {
    const employeeItem = { key: '/employee/query', icon: <SearchOutlined />, label: '员工查询' };
    return [{ key: employeeItem.key, icon: employeeItem.icon, label: employeeItem.label }];
  }
  
  // 2. 置顶项：快速融合（per D-02）
  const items: MenuProps['items'] = [];
  if (topItem.roles.includes(userRole)) {
    items.push({ key: topItem.key, icon: topItem.icon, label: topItem.label });
  }
  
  // 3. 各分组 SubMenu
  for (const group of groups) {
    // 管理组需要插入飞书菜单项
    let groupChildren = [...group.children];
    if (group.key === 'group-admin' && feishuItems.length > 0) {
      groupChildren = [...groupChildren, ...feishuItems];
    }
    
    const visibleChildren = groupChildren
      .filter(child => child.roles.includes(userRole))
      .map(child => ({ key: child.key, icon: child.icon, label: child.label }));
    
    if (visibleChildren.length > 0) {
      items.push({
        key: group.key,
        icon: group.icon,
        label: group.label,
        children: visibleChildren,
      });
    }
  }
  
  return items;
}
```
